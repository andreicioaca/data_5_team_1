select  
	od.OrderID ,
	od.UnitPrice ,
	od.Quantity ,
	round ( od.unitprice * od.Quantity, 2) as SoldAMT,
	round ( od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity, 2) as SoldAMTfp,
	od.Discount,
	o.freight,
	round ( od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity - o.Freight, 2) as Profit ,
	o.ShipRegion,
	CASE 
	when od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity < (( select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od) /2)
	then 'low sale'
	when od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity > (( select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od) /2) and 
		od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity < (( select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od)) + (( select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od) /2)
	then 'medium sale'
	when od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity >((select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od) + ( select avg(od.unitprice * od.Quantity - od.discount * od.unitprice * od.Quantity) from "Order Details" od) /2)
	then 'high sale' END as 'sale category'
	from "Order Details" od 
	join Orders o on od.OrderID = o.OrderID
	where o.OrderDate  like '2016%'
	order by soldamtfp

Aristide Irakoze: 
===============

1. Details about Each Order including Countries and Prices

SELECT  OrderDate, ShippedDate , ShipName , ShipCountry , Quantity, UnitPrice, Discount
FROM Orders o JOIN 'Order Details' od ON od.OrderID = o.OrderID

2.Money from Each Order

SELECT  OrderDate, ShippedDate , ShipName , ShipCountry ,
Quantity * (UnitPrice - Discount ) as Amount
FROM Orders o
JOIN 'Order Details' od ON od.OrderID = o.OrderID

3. Money from Each Order only in 2016

SELECT  OrderDate, ShippedDate , ShipName , ShipCountry ,
Quantity * (UnitPrice - Discount ) as Cost FROM Orders o
JOIN 'Order Details' od ON od.OrderID = o.OrderID
WHERE  ShippedDate LIKE '2016%'

4. Total Money Per Country in  2016

SELECT  OrderDate, ShippedDate  , ShipCountry ,
SUM ( Quantity * (UnitPrice - Discount ) ) as Total FROM Orders o
JOIN 'Order Details' od ON od.OrderID = o.OrderID
GROUP BY ShipCountry
HAVING  ShippedDate LIKE '2016%'
